void qsort(int *nums, int left,int right){
    int pivot = nums[0];
    int l=0;
    int r=right;
    while(l<r){
        while(nums[l]<pivot){
            l++;
        }
        while(nums[r]>=pivot){
           r--;
        }
        if(l<r){
            swap(nums,l,r);
        }
    }
    
    qsort(nums,left,l);
    qsort(nums,l+1,right);
}
